175 	/**
176 	 * Rewrites the whole input CKR to global and local programs.
177 	 */
178 	public void rewrite(){
179 		//SROEL2DatalogRewriter rewriter = new SROEL2DatalogRewriter();
180 		
181 		long startTime = System.currentTimeMillis();
182 		
183 		//Rewriting for global context.
184 		RLGlobal2DatalogRewriter globalrewriter = new RLGlobal2DatalogRewriter();
185 		datalogGlobal = globalrewriter.rewrite(inputCKR.getGlobalOntology());
186 		//System.out.println("Rewriting program for global context complete.");
187 		
188 		//Computation of set of contexts and associations to modules
189 		computeSets(datalogGlobal);
190 		//System.out.println("Set of contexts and modules associations computed.");
191 		
192 		//Computation of local contexts knowledge bases.
193 		computeLocalKB();
194 		
195 		//Rewriting for local contexts knowledge bases.
196 		RLLocal2DatalogRewriter localrewriter = new RLLocal2DatalogRewriter();
197 		
198 		datalogLocal = new LinkedList<DLProgram>();
199 		for (String c : contextsSet) {
200 			//System.out.println("Rewriting program for " + c.replaceAll("\"", ""));
201 			localrewriter.setContextID(c.replaceAll("\"", ""));
202 			datalogLocal.add(localrewriter.rewrite(contextsOntologies.get(c)));
203 		}
204 		
205 		//for (OWLOntology o : inputCKR.getLocalOntology()) {
206 		//	localrewriter.setLocalID("c");
207 		//	//o.getOWLOntologyManager().addAxioms(o, inputCKR.getGlobalOntology().getAxioms());
208 		//	
209 		//	datalogLocal.add(localrewriter.rewrite(o));
210 		//	//System.out.println(datalogLocal.getLast().getStatements().size());
211 		//}
212 				
213 		long endTime = System.currentTimeMillis();
214 		rewritingTime = endTime - startTime;
215 
216 		//System.out.println("Rewriting completed in " + rewritingTime + " ms.");
217 		//System.out.println(datalogGlobal.getStatements().size());
218 		
219 		//Compute CKR Program.
220 		datalogCKR = new DLProgram();
221 	
222 		datalogCKR.addAll(datalogGlobal.getStatements());
223 		for (DLProgram dlProgram : datalogLocal) {
224 			datalogCKR.addAll(dlProgram.getStatements());	
225 		}
226 		//Add local inference rules.
227 		datalogCKR.addAll(DeductionRuleset.getPloc());
228 		//Add local propagation rules.
229 		datalogCKR.addAll(DeductionRuleset.getPd());
230 		
231 		//StringReader reader = new StringReader("hasModule(X,Y) :- triple(X, \"hasModule\", Y, \"g\").");
232 		//DLProgramParser dlProgramParser = new DLProgramParser(reader);
233 		//try {
234 		//	datalogCKR.addAll(dlProgramParser.program().getStatements());
235 		//} catch (ParseException e) {
236 		//	e.printStackTrace();
237 		//}
238 	}

252

255 		
256 		//Compute DKB Program.
257 		datalogCKR = new DLProgram();
258 	
259 		datalogCKR.addAll(datalogGlobal.getStatements());

267 	/**
268 	 * Interacts with DLV to compute the set of contexts
269 	 * and their associations to knowledge modules.
270 	 * 
271 	 * @param datalogGlobal global program to be evaluated by DLV
272 	 */
273 	private void computeSets(DLProgram datalogGlobal){
274 		
275 		DLVInvocation invocation = DLVWrapper.getInstance().createInvocation(dlvPath);
276 		DLVInputProgram inputProgram = new DLVInputProgramImpl();
277 
278 		try {			
279 			DLProgramStorer storer = new DLProgramStorerImpl();
280 			StringBuilder target = new StringBuilder();
281 			storer.store(datalogGlobal, target);
282             
283 			//Add to DLV input program the contents of global program. 
284 			String datalogGlobalText = target.toString();
285 			inputProgram.addText(datalogGlobalText);
286 			
287 			//inputProgram.addText("triple(c1,\"hasModule\",m1,\"g\")." + 
288 			//                     "  inst(c1,\"Context\",\"g\")."+ 
289 			//                     "triple(c1,\"hasModule\",m2,\"g\")." + 
290 			//                     "triple(X, \"hasModule\", m3, \"g\") :- inst(X,\"Context\",\"g\").");
291 			
292 			//Set input program for current invocation.
293 			invocation.setInputProgram(inputProgram);
294 			
295 			//Filter for \triple and \inst predicates. 
296 			//System.out.println(inputProgram.getCompleteText());
297 			List<String> filters = new LinkedList<String>();
298 			filters.add("tripled");
299 			filters.add("instd");
300 			invocation.setFilter(filters, true);
301 			
302 			//List of computed models, used to check at least a model is computed.
303 			final List<Model> models = new ArrayList<Model>();
304 			
305 			//Model handler: retrieves contexts and associations in the computed model(s).
306 			invocation.subscribe(new ModelHandler() {
307 
308 				@Override
309 				public void handleResult(DLVInvocation paramDLVInvocation,
310 						ModelResult modelResult) {
311 					
312 					//System.out.print("{ ");
313 					Model model = (Model) modelResult;
314 					models.add(model);
315 
316 					//model.beforeFirst();
317 					//while (model.hasMorePredicates()) {}
318 
319 					//Predicate predicate = model.nextPredicate();
320 					Predicate predicate = model.getPredicate("instd");
321 					if (predicate != null){
322 						//System.out.println(predicate.name() + ": ");
323 						while (predicate.hasMoreLiterals()) {
324 
325 							Literal literal = predicate.nextLiteral();
326 							if (literal.getAttributeAt(1).toString().equals("\"Context\"")) {
327 								
328 								//Add context to list of inferred contexts.
329 								contextsSet.add(literal.getAttributeAt(0).toString());
330 							
331 								//System.out.print(literal);
332 								//System.out.println(", ");	
333 							}
334 						}
335 					}
336 					
337 					predicate = model.getPredicate("tripled");
338 					if (predicate != null){
339 						//System.out.println(predicate.name() + ": ");
340 						while (predicate.hasMoreLiterals()) {
341 
342 							//Add module association for each context.
343 							Literal literal = predicate.nextLiteral();
344 							if (literal.getAttributeAt(1).toString().equals("\"hasModule\"")) {
345 									
346 								String[] association = new String[2];
347 								association[0] = literal.getAttributeAt(0).toString();
348 								association[1] = literal.getAttributeAt(2).toString();
349 								hasModuleAssociations.add(association);
350 							
351 								//System.out.print(literal);
352 								//System.out.println(", ");	
353 							}
354 						}
355 					}
356 					
357 					//System.out.println("}");
358 					//System.out.println();
359 				}
360 			});
361 			
362 			long startTime = System.currentTimeMillis();
363 			
364 			invocation.run();
365 			invocation.waitUntilExecutionFinishes();
366 			
367 			long endTime = System.currentTimeMillis();
368 			globalModelComputationTime = endTime - startTime;
369 			
370 			//System.out.println("Global computation time: " + globalModelComputationTime + " ms.");
371 			
372 			List<DLVError> k = invocation.getErrors();
373 			if (k.size() > 0)
374 				System.err.println(k);
375 			
376 			//System.out.println("Number of computed models: " + models.size());
377 			if(models.size() == 0) 
378 				System.err.println("[!] No models for global context program.");
379 			
380 			//for (String[] a : hasModuleAssociations) {
381 			//	System.out.println(a[0] + " -> " + a[1]);
382 			//}
383 			
384 			//System.out.println("Contexts: ");
385 			//for (String s : contextsSet) {
386 			//	System.out.println(s);
387 			//	for(String[] a : hasModuleAssociations){
388 			//		if(a[0].equals(s))
389 			//		System.out.println("  -> " + a[1]);	
390 			//	}
391 			//}
392 		} catch (DLVInvocationException | IOException e) {
393 			e.printStackTrace();
394 		}
395 	}

399 	/**
400 	 * Associates a local knowledge base to each of the contexts,
401 	 * by adding the associated knowledge modules (and the global knowledge 
402 	 * from global).
403 	 * 
404 	 */
405 	private void computeLocalKB(){
406 		for (String s : contextsSet) {
407 			//System.out.println(s);
408 			
409 			OWLOntology contextOntology = null;
410 			try {
411 				contextOntology = manager.createOntology();
412 				
413 				for(String[] a : hasModuleAssociations){
414 					if(a[0].equals(s)){
415 						//System.out.println("  -> " + a[1]);
416 					
417 						//Check that to each module name a local ontology exists!
418 						//int moduleIndex = inputCKR.getLocalOntologyName().indexOf(a[1]);
419 						CKRModule mod = inputCKR.getLocalModule(a[1]);
420 					
421 						if (mod == null) {
422 							System.err.println("[!] Warning: no ontology associated to module " + a[1]);
423 						} else {
424 							manager.addAxioms(contextOntology, 
425 									//inputCKR.getLocalOntology().get(moduleIndex).getAxioms()
426 									mod.getModuleOntology().getAxioms());
427 							//System.out.println("Module " + a[1] + " added to " + a[0]);
428 						}
429 					}
430 				}
431 				
432 				//***DO NOT*** Add axioms from global ontology.
433 				//manager.addAxioms(contextOntology, inputCKR.getGlobalOntology().getAxioms());
434 			} catch (OWLOntologyCreationException e) {
435 				e.printStackTrace();
436 			} 
437 			//if ((contextOntology == null) || (contextOntology.isEmpty()))
438 			//System.out.println("Warning: ontology for " + s + " is empty.");
439 			
440 			contextsOntologies.put(s, contextOntology);
441 		}
442 	}
