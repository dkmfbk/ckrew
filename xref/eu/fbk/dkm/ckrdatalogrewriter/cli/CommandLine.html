1   package eu.fbk.dkm.ckrdatalogrewriter.cli;
2   
3   import it.unical.mat.wrapper.DLVInputProgram;
4   import it.unical.mat.wrapper.DLVInvocationException;
5   
6   import java.io.IOException;
7   
8   import org.semanticweb.drew.dlprogram.parser.ParseException;
9   import org.semanticweb.owlapi.model.OWLOntology;
10  import org.semanticweb.owlapi.model.OWLOntologyCreationException;
11  
12  import eu.fbk.dkm.ckrdatalogrewriter.rl.cli.CKRDReWRLCLI;
13  
14  /**
15   * @author Loris 
16   * @version 1.3 
17   * 
18   * Abstract class for command line interface.
19   * (Extended from DreW CLI).
20   */
21  public abstract class CommandLine {
22  
23  	//--- ABSTRACT METHODS ------------------------------------
24  	/**
25  	 * Abstract method for parsing of command line parameters.
26  	 * 
27  	 * @param args command line parameters
28  	 * @return <code>true</code> if call conforms with usage, <code>false</code> otherwise. 
29  	 */
30  	public abstract boolean parseArgs(String[] args);
31  
32  	/**
33  	 * Abstract method for the main execution thread of the application. 
34  	 */
35  	public abstract void go();
36  
37  	/**
38  	 * Abstract method for handling and execution of input Conjunctive Query.<br/>
39  	 * (Not currently implemented).
40  	 *  
41  	 * @param ontology
42  	 * @param inputProgram 
43  	 */
44  	public abstract void handleCQ(OWLOntology ontology, DLVInputProgram inputProgram);
45  	
46  	//--- MAIN ------------------------------------------------
47  	/**
48  	 * @param args
49  	 * @throws DLVInvocationException
50  	 * @throws ParseException
51  	 * @throws IOException
52  	 * @throws OWLOntologyCreationException
53  	 */
54  	public static void main(String[] args) throws OWLOntologyCreationException,
55  			IOException, ParseException, DLVInvocationException {
56  		
57  		if (args.length == 0) {
58  			String usage = "Usage: ckrew <global-context-file> [<local-module-file> | <options>]";
59  			System.err.println(usage);
60  			System.exit(0);
61  		} else {
62  			CKRDReWRLCLI.main(args);
63  		}		
64  	}
65  
66  }
