58  	@Override
59  	public void go() {
60  		
61  		//Create a new CKR knowledge base to manage input ontologies
62  		inputCKR = new CKRKnowledgeBase();
63  		
64  		//Parse input for global and local (modules) ontologies
65  		printBanner();
66  		if (!parseArgs(args)) {
67  			printUsage();
68  			System.exit(1);
69  		}
70  		
71  		//Check compatibility of options
72  		if(isDLlite && isMRCKR) {
73  			System.err.println("[!] Incompatible options: -dllite, -mrckr");
74  			System.exit(1);			
75  		}		
76  		
77  		//Check global input file existence
78  		File ontofile = new File(inputCKR.getGlobalOntologyFilename());
79  		if(!ontofile.exists()){
80  			System.err.println("[!] Input global ontology file does not exists.");
81  			System.exit(1);
82  		}
83  		
84  		//Manage TRIG files.
85  		if(trigInput) parseTrigFile();
86  
87  		//Check local input file existence
88  		for(CKRModule mod : inputCKR.getLocalModule()){
89  			File modfile = new File(mod.getModuleFilename());
90  			
91  			if(!modfile.exists()){
92  				System.err.println("[!] Input local ontology file does not exist: " + mod.getModuleFilename());
93  				System.exit(1);
94  			}		
95  		}	
96  		
97  		//Load ontology files.
98  		//TODO: unify MR-CKR case in loadOntologies (kept separated for debugging)
99  		if(isMRCKR) 
100 			try {
101 			inputCKR.loadOntologiesMR();
102 			} catch (Exception e) {
103 				throw new RuntimeException(e);
104 		} else 
105 			try {
106 			inputCKR.loadOntologies();
107 		} catch (Exception e) {
108 			throw new RuntimeException(e);
109 		}
110 
111 		
112 		if(verbose) System.out.println("Global and local ontologies loaded.");
113 		
114 		if(verbose) System.out.println("Number of input logical axioms: " + inputCKR.getCKRSize());
115 		
116 		if(!isDLlite){//TODO: check profile for OWL-QL ontologies
117 			
118 			//Check if profile of ontologies is in the correct fragment.
119 			OWLProfileReport report = inputCKR.checkProfiles();
120 			if (!report.isInProfile()) {
121 				System.err.println(report);
122 				System.exit(1);
123 			} 
124 			if(verbose) System.out.println("Global and local ontologies in OWL-RL.");
125 		}
126 				
127 		//Handle translation of input CKR
128 		handleOntology(inputCKR);
129 	}

132     /**
133      * Handles the case in which transformation to Datalog 
134      * of the input CKR is required. 
135      * 
136      * @param ckr input CKR knowledge base
137      * */
138 	private void handleOntology(CKRKnowledgeBase ckr) {
139 		
140 		//Create new program for the input CKR.
141 		outputCKRProgram = new CKRProgram(ckr);
142 				
143 		//Set possibly custom DLV path
144 		if(dlvPath != null) 
145 			outputCKRProgram.setDlvPath(dlvPath);
146 		else{
147 			dlvPath = DEFAULT_DLV_PATH;
148 			outputCKRProgram.setDlvPath(dlvPath);			
149 		}
150 		
151 		//Set possibly custom output file path
152 		if(outputFilePath != null) 
153 			outputCKRProgram.setOutputFilePath(outputFilePath);
154 		else{
155 			outputFilePath = DEFAULT_OUTPUT_FILENAME;
156 			outputCKRProgram.setOutputFilePath(outputFilePath);			
157 		}
158 		
159 		//Rewrite the program.
160 		if(verbose) System.out.println("Rewriting program...");
161 		if(isDLlite)
162 			outputCKRProgram.rewriteDLR(); //option: DLliteR DKB
163 		else if(isMRCKR)
164 			outputCKRProgram.rewriteMRCKR(); //option: MR-CKR on OWL-RL
165 		else
166 			outputCKRProgram.rewrite(); //default: CKR on OWL-RL
167 		
168 		//XXX: #####################
169 		//TODO: ### Work su CKRProgram per gestire translation MR-CKR! ###
170 		
171 //		if(verbose && isMRCKR) {
172 //			System.out.println("MR-CKR test complete.");
173 //			System.exit(0);			
174 //		}
175 			
176 		if(verbose && !isDLlite) {
177 			
178 			if(!isMRCKR)
179 				System.out.println("Global model computation time: " + outputCKRProgram.getGlobalModelComputationTime() + " ms.");
180 			
181 			System.out.println("Set of contexts and modules associations computed:");
182 			for (String s : outputCKRProgram.getContextsSet()) {
183 				System.out.println(s);
184 				for(String[] a : outputCKRProgram.getHasModuleAssociations()){
185 					if(a[0].equals(s))
186 						System.out.println("  -> " + a[1]);
187 				}
188 			}
189 			System.out.println("Rewriting completed in " + outputCKRProgram.getRewritingTime() + " ms.");
190 			
191 			System.out.println("Program size (number of statements): " + outputCKRProgram.getProgramSize());
192 		}
193 		
194 		//Store the program to file.
195 		try {
196 			outputCKRProgram.storeToFile();
197 			if(verbose) System.out.println("CKR program saved in: " + outputCKRProgram.getOutputFilePath() + "\n");
198 		} catch (Exception e) {
199 			e.printStackTrace();
200 		}
201 		System.out.println("CKRew: process complete.");
202 	}

210 	@Override
211 	/** 
212 	 * Parses the input parameters. 
213 	 * 
214 	 * @param args input parameter string
215 	 * */
216 	public boolean parseArgs(String[] args) {
217 		
218 		if (args.length == 0) {			
219 			System.err.println("[!] Missing argument: <global-context-file>\n");
220 			System.err.println();			
221 			return false;
222 			
223 		} else {
224 			inputCKR.setGlobalOntologyFilename(args[0]);
225 			if(verbose) System.out.println("Global ontology: " + inputCKR.getGlobalOntologyFilename());
226 			
227 			for (int i = 1; i < args.length; i++) {				
228 				switch (args[i]) {
229 				case "-dlv":
230 					if(i+1 < args.length && !args[i+1].startsWith("-"))
231 					   dlvPath = args[++i];
232 					else {
233 						System.err.println("[!] Missing argument: <dlv-path>");						
234 						return false;
235 					}					
236 					break;
237 				case "-out":
238 					if(i+1 < args.length && !args[i+1].startsWith("-"))
239 					   outputFilePath = args[++i];
240 					else {
241 						System.err.println("[!] Missing argument: <output-file>");						
242 						return false;
243 					}					
244 					break;
245 				case "-v":
246 				    verbose = true;
247 					break;
248 				case "-trig":
249 				    trigInput = true;
250 					break;
251 				case "-dllite":
252 				    isDLlite = true;
253 				    if(verbose) System.out.println("DLliteR input mode.");
254 					break;					
255 				case "-mrckr":
256 				    isMRCKR = true;
257 				    if(verbose) System.out.println("MR-CKR input mode.");
258 					break;										
259 				default:
260 					if(trigInput) {
261 						return false; //more arguments than expected... 
262 					} else {//add local module to input CKR
263 						inputCKR.addLocalOntologyWithFilename(args[i]);
264 						if(verbose) System.out.println("Local ontology: " + args[i]);					
265 					break;
266 					}
267 				}
268 			}
269 			return true;
270 		}
271 	}

273 	/**
274 	 * Split and uses the TRIG file in input. 
275 	 * 
276 	 */
277 	void parseTrigFile() {
278 		
279 		String trigFilename = inputCKR.getGlobalOntologyFilename();
280 		
281 		TrigSplitter splitter = new TrigSplitter(trigFilename);
282 		try {
283 			splitter.split();
284 			splitter.saveToTurtleFiles();
285 			
286 			//Set global module
287 			inputCKR.setGlobalOntologyFilename(splitter.getTrigFolder() + "global.n3");
288 			if(verbose) System.out.println("Global turtle: " + inputCKR.getGlobalOntologyFilename());
289 			
290 			//Set local modules
291 			for (String name : splitter.getOutputFileNames()) {
292 				if (!name.contains("global")){
293 					inputCKR.addLocalOntologyWithFilename(splitter.getTrigFolder() + name);
294 					if(verbose) System.out.println("Local module:" + splitter.getTrigFolder() + name);
295 				}
296 			}
297 			
298 		} catch (IOException e) {
299 			e.printStackTrace();
300 		}		
301 		return;
302 	}

315 	 * Prints usage of the command line and exits.
316 	 */
317 	void printUsage() {
318 
319 		String usage = 
320 				"Usage: ckrew <global-context-file> [<local-module-file> | <options>]\n"
321 				+ //
322 				" <global-context-file>\n"
323 				+ //
324 				"  Ontology file containing the global context for the input CKR. \n"
325 				+ //
326 				" <local-module-file>\n"
327 				+ //
328 				"  Ontology files (zero or more) containing a knowledge module for the input CKR. \n"
329 				+ //
330 				"Example: ckrew global.n3 m1.n3 m2.n3\n"
331 				+ //
332 				"Options:\n"
333 				+ //
334 				" -mrckr: interprets input as multi-relational sCKR\n"
335 				+ //
336 				" -dllite: interprets global ontology as (single context) DLliteR defeasible KB\n"
337 				+ //
338 				" -v: verbose (prints more information about loading and rewriting process)\n"
339 				+ //
340 				" -out <output-file>: specifies the path to the output program file (default: output.dlv)\n"
341 				+ //
342 				" -trig: specifies that the input is provided as a single TRIG file\n"
343 		        + //
344 		        " -dlv <dlv-path>: specifies the path to the DLV executable (default: localdlv/dlv)\n";
345 
346 		System.out.println(usage);
347 	}
